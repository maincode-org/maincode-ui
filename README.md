## Install

```bash
npm install --save maincode-ui
```

## Usage

```tsx
import React, { Component } from 'react';

import MyComponent from 'maincode-ui';
import 'maincode-ui/dist/index.css';

class Example extends Component {
  render() {
    return <MyComponent />;
  }
}
```
